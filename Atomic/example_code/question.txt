func question(message) {
	
	set results >> prompt(message, " Y/n: ")
	
	if(toLower(results) = "y") {
		return true;
	}
	else if(toLower(results) = "n") {	
		return false;
	}
	else {	
		return null;
	}
}

set isRobot >> question("are you a robot?")
write(isRobot)
if(isRobot) {
	write("you are a robot ):")
}

else if(isRobot = false) {
	write("Yay you are not a robot :)")
}
